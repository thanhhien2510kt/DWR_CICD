    function timeAgo(date) {

        const seconds = Math.floor((new Date() - parsedDate) / 1000); // Tính s? giây ?ă qua

        // X? lý theo th? t? các kho?ng th?i gian t? l?n ??n nh?
        for (const [key, value] of Object.entries(intervals)) {

                function formatThousandsSeparator(number) {
                const parts = number.toString().split('.');
                const wholePart = parts[0].replace(/(\d)(?=(\d{3})+$)/g, '$1.'); // Add commas for thousands
                const decimalPart = parts[1] || ''; // Handle decimal part
                return wholePart + (decimalPart ? `.${decimalPart}` : '');

            }

        function onChangeTypeNumber(input) {
            try {
                var value = input.value.replace(/\./g, "");
                if (!isNaN(value))
                    input.value = value == "" ? "" : formatThousandsSeparator(parseFloat(value));
            } catch (e) {
                console.log('err: ' + e)
            }

        }
